18template <typename T>
19void selectionSort(orderedLinkedList<T*>& arr, function<bool(const T*, const T*)> compare)
20{
21 int n = arr.size();
22 for (int i = 0; i < n - 1; ++i)
23 {
24 int minIndex = i;
25 for (int j = i + 1; j < n; ++j)
26 {
27 T* a = arr.get(j);
28 T* b = arr.get(minIndex);
29 if (a && b && compare(a, b))
30 {
31 minIndex = j;
32 }
33 }
34 if (minIndex != i)
35 {
36 T* tempA = arr.get(i);
37 T* tempB = arr.get(minIndex);
38 if (tempA && tempB)
39 {
40 std::swap(*tempA, *tempB);
41 }
42 }
43 }
44}
